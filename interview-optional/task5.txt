a) SELECT COUNT(*) AS total_transactions FROM TRANSACTION t
   JOIN ACCOUNT a ON t.ACCOUNT_ID = a.ID
   JOIN USER u ON a.USER_ID = u.ID
   WHERE u.ID = '356789' AND t.CATEGORY_ID = '15' AND YEAR(t.POSTED_DATE) = 2013;

b) SELECT t.ID, t.ACCOUNT_ID, t.CATEGORY_ID, t.NAME, t.AMOUNT, t.POSTED_DATE FROM TRANSACTION t
   JOIN ACCOUNT a ON t.ACCOUNT_ID = a.ID
   WHERE a.USER_ID = '<userId>'
     AND a.ID = '<accountId>'
     AND t.POSTED_DATE >= DATE_TRUNC('month', CURRENT_DATE)
     AND t.POSTED_DATE < DATE_TRUNC('month', CURRENT_DATE) + INTERVAL '1 month';

c) SELECT t.ID, t.ACCOUNT_ID, t.CATEGORY_ID, t.NAME, t.AMOUNT, t.POSTED_DATE FROM TRANSACTION t
   JOIN ACCOUNT a ON t.ACCOUNT_ID = a.ID
   WHERE a.USER_ID = '<userId>'
     AND t.AMOUNT > 0
     AND t.POSTED_DATE >= DATE_TRUNC('month', CURRENT_DATE)
     AND t.POSTED_DATE < DATE_TRUNC('month', CURRENT_DATE) + INTERVAL '1 month';
